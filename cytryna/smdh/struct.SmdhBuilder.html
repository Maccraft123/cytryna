SMDH builder

Examples

use cytryna::prelude::*;

let bmp_image = bmp::Image::new(48, 48);
let smdh = Smdh::builder()
    .with_short_desc("An example")?
    .with_long_desc("This is an example data")?
    .with_publisher("Maya")?
    .with_icon((&bmp_image).try_into()?)
    .build()?;
